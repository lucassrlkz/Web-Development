CREATE DATABASE BdAcervo;
USE BdAcervo;

CREATE TABLE USUARIOS(
ID INT PRIMARY KEY AUTO_INCREMENT,
NOME VARCHaR(50)NOT NULL,
EMAIL VARCHAR(150)NOT NULL,
LOGIN VARCHAR(15)NOT NULL,
SENHA VARCHAR(50)NOT NULL);

CREATE TABLE LIVROS(
ISBN INT PRIMARY KEY,
TITULO VARCHAR(50)NOT NULL,
EDICAO VARCHAR(10)NOT NULL,
CATEGORIA VARCHAR(10)NOT NULL,
PRECO FLOAT NOT NULL,
ANOPUBLICACAO VARCHAR(6)NOT NULL,
IMAGEM VARCHAR(100));

CREATE TABLE AUTORES(
ID INT PRIMARY KEY AUTO_INCREMENT,
NOME VARCHAR(50)NOT NULL,
NACIONALIDADE VARCHAR(15)NOT NULL,
ISBN_L INT, FOREIGN KEY (ISBN_L)REFERENCES LIVROS(ISBN));

CREATE TABLE RESERVADOS(
ID INT PRIMARY KEY AUTO_INCREMENT,
ID_LIVRO INT, FOREIGN KEY(ID_LIVRO)REFERENCES LIVROS(ISBN),
ID_USUARIO INT, FOREIGN KEY(ID_USUARIO)REFERENCES USUARIOS(ID) 
);
